module up_down_counter (
    input clk,
    input reset,
    input up_down,    // 1: Up, 0: Down
    input enable,
    output reg [3:0] count
);
    always @(posedge clk) begin
        if (reset)
            count <= 4'b0000;
        else if (enable) begin
            if (up_down)
                count <= count + 1;
            else
                count <= count - 1;
        end
    end

       initial begin
        $dumpfile("dump.vcd");
        $dumpvars(0, up_down_counter);
    end
endmodule
